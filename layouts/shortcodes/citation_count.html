{{- $doi := .Get "doi" -}} {{- $label := or (.Get "label") "Cited by" -}} {{- if not $doi -}} citation unavailable {{- else -}} {{- $url := printf "https://api.openalex.org/works/https://doi.org/%s" $doi -}} {{- with try (resources.GetRemote $url) -}} {{- with .Err -}} citation unavailable {{- else with .Value -}} {{- $work := . | transform.Unmarshal -}} {{- with $work.cited_by_count -}} | {{ $label }} {{ . }} {{- else -}} citation unavailable {{- end -}} {{- end -}} {{- end -}} {{- end -}}
